def calculate_composite_score(components: Dict[str, float], weights: Dict[str, float]) -> float:
    valid_components = {k: v for k, v in components.items() if k in weights and v is not None}
    if not valid_components:
        return 0.5  # Neutral fallback
    
    total_weight = sum(weights[k] for k in valid_components)

    return weighted_sum / total_weight
